@app.route('/register', methods=['GET', 'POST']) def register(): if request.method == 'POST': username = request.form['username'] email = request.form['email'] passport = request.form['passport'] password = request.form['password'] # Provera da li korisnik postoji existing_user = User.query.filter_by(username=username).first() if existing_user: return "User already exists!" hashed_password = generate_password_hash(password) new_user = User( username=username, email=email, passport=passport, password=hashed_password, balance=0 ) db.session.add(new_user) db.session.commit() return redirect('/login') return render_template('register.html')
